# This is bold

This is a bold **text**
<!-- Double asterisks make text bold -->

This is a bold __text__
<!-- Double underscores also make text bold -->

thisis**bold**text
<!-- Bold formatting can be used inside words -->


# This is italic

This is an italic *text*
<!-- Single asterisks make text italic -->

This is an italic _text_
<!-- Single underscores also make text italic -->

thisis*italic*text
<!-- Italic formatting can be used inside words -->


# This is bold and italic

This is ***bold and italic*** text
<!-- Triple asterisks make text bold and italic -->

This is **_bold and italic_** text
<!-- Combining bold and italic syntax -->

This is __*bold and italic*__ text
<!-- Another valid way to combine bold and italic -->

thisis***boldAndItalic***text
<!-- Bold and italic formatting can also be used inside words -->
